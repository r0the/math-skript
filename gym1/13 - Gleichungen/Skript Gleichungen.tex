    scale=#1,
    very thick,
    balance style/.style={draw=blue, fill=lightblue},
    block style/.style={draw=red, fill=lightred}
  ]
  % Aufhängungen der Schalen

  \draw[blue,ultra thick] (-4,1.4) -- (-6,4);
